using System;
using System.Collections.Generic;
using System.Text;

namespace grocwry
{
    public class Order : Products
    {
        private double totalPrice;

        public Order() : base() { }
        public Order(string i, double p, int q) : base(i, p, q) { }

        public double GetTotalPrice()
        {
            CalcTotalPrice();
            return totalPrice;
        }
        private void CalcTotalPrice()
        {
            totalPrice = GetPrice() * GetQuantity();
        }

        public void ToString()
        {
            Console.WriteLine(GetItemName() + " " + GetPrice() + " x " + GetQuantity() + " " + GetTotalPrice());
        }
    }
}
